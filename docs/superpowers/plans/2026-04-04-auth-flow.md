# Auth Flow Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Add Satoshi font, glassmorphic auth screens at `/`, move WalletBuilder to `/wallet`, and build a reusable `<VirtualCard />` component integrated into the wallet.

**Architecture:** Option B — thin server-component shells, all logic in focused client components. `AuthForm.tsx` owns login/signup state and API calls. `VirtualCard.tsx` is a pure presentational component. `WalletBuilder.tsx` is updated to read `user_id` from localStorage and render VirtualCards.

**Tech Stack:** Next.js 16 App Router, React 19, TypeScript, Tailwind CSS v4, `@fontsource/satoshi`, `sonner` toasts, `next/navigation` useRouter.

```tsx
import { cn } from "@/lib/utils";

const BANK_COLORS: Record<string, { bg: string; text: string }> = {

  "ICICI BANK":          { bg: "#6b2d1a", text: "#ffb99a" },

  "SBI CARD":            { bg: "#1a5c6b", text: "#67e8f9" },
  "KOTAK MAHINDRA BANK": { bg: "#6b1a5c", text: "#f0abfc" },
  "AMERICAN EXPRESS":    { bg: "#1a4b6b", text: "#93c5fd" },
  "YES BANK":            { bg: "#5c6b1a", text: "#d9f99d" },
  "INDUSIND BANK":       { bg: "#6b4b1a", text: "#fde68a" },
};

function getBankStyle(bankName: string) {
  return BANK_COLORS[bankName.toUpperCase()] ?? { bg: "#2a2a40", text: "#a5b4fc" };
}

function getBankInitials(bankName: string): string {
  return bankName
    .split(" ")
    .filter((w) => !["BANK", "CARD", "OF"].includes(w.toUpperCase()))
    .slice(0, 2)
    .map((w) => w[0])
    .join("")
    .toUpperCase();
}

interface VirtualCardProps {
  cardName: string;
  bankName: string;
  selected?: boolean;
  onClick?: () => void;
  compact?: boolean;
  className?: string;
}

export function VirtualCard({
  cardName,
  bankName,
  selected = false,
  onClick,
  compact = false,
  className,
}: VirtualCardProps) {
  const { bg, text } = getBankStyle(bankName);
  const initials = getBankInitials(bankName);

  return (
    <div
      onClick={onClick}
      role={onClick ? "button" : undefined}
      tabIndex={onClick ? 0 : undefined}
      onKeyDown={onClick ? (e) => e.key === "Enter" && onClick() : undefined}

        "relative w-full overflow-hidden rounded-2xl transition-all duration-200",
```

- [ ] **Step 1: Create the file**

Create `src/components/AuthForm.tsx`:

```tsx
"use client";

import { useState } from "react";
import { useRouter } from "next/navigation";
import { toast } from "sonner";
import { Loader2, Eye, EyeOff } from "lucide-react";
import { cn } from "@/lib/utils";

const API_BASE = "http://localhost:5000";

type Mode = "login" | "signup";

interface SignupPayload {
  firstName: string;
  lastName: string;
  email: string;
  password: string;
}

interface LoginPayload {
  identifier: string;
  password: string;
}

export default function AuthForm() {
  const router = useRouter();
  const [mode, setMode] = useState<Mode>("login");
  const [isLoading, setIsLoading] = useState(false);
  const [showPassword, setShowPassword] = useState(false);
  const [showConfirm, setShowConfirm] = useState(false);

  // Signup fields
  const [firstName, setFirstName] = useState("");
  const [lastName, setLastName] = useState("");
  const [signupEmail, setSignupEmail] = useState("");
  const [signupPassword, setSignupPassword] = useState("");
  const [confirmPassword, setConfirmPassword] = useState("");

  // Login fields
  const [identifier, setIdentifier] = useState("");
  const [loginPassword, setLoginPassword] = useState("");

  function switchMode(next: Mode) {
    setMode(next);
    setShowPassword(false);
    setShowConfirm(false);
  }

  async function handleSignup(e: React.FormEvent) {
    e.preventDefault();
    if (signupPassword !== confirmPassword) {
      toast.error("Passwords do not match.");
      return;
    }
    setIsLoading(true);
    try {
      const res = await fetch(`${API_BASE}/api/auth/signup`, {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          firstName,
          lastName,
          email: signupEmail,
          password: signupPassword,
        } satisfies SignupPayload),
      });
      const data = await res.json();
      if (!res.ok) {
        toast.error(data.error ?? "Signup failed. Please try again.");
        return;
      }
      localStorage.setItem("zuno_user_id", data.user_id);
      toast.success(`Welcome to Zuno, ${firstName}!`);
      router.push("/wallet");
    } catch {
      toast.error("Could not reach the server. Is the backend running?");
    } finally {
      setIsLoading(false);
    }
  }

  async function handleLogin(e: React.FormEvent) {
    e.preventDefault();
    setIsLoading(true);
    try {
      const res = await fetch(`${API_BASE}/api/auth/login`, {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          identifier,
          password: loginPassword,
        } satisfies LoginPayload),
      });
      const data = await res.json();
      if (!res.ok) {
        toast.error(data.error ?? "Login failed. Please try again.");
        return;
      }
      localStorage.setItem("zuno_user_id", data.user_id);
      toast.success(`Welcome back, ${data.firstName}!`);
      router.push("/wallet");
    } catch {
      toast.error("Could not reach the server. Is the backend running?");
    } finally {
      setIsLoading(false);
    }
  }

  const inputClass = cn(
    "w-full h-11 rounded-xl px-4 text-sm text-white",
    "bg-white/5 border border-white/10",
    "placeholder:text-white/30",
    "focus:outline-none focus:border-[#03FDFC]/40 focus:ring-1 focus:ring-[#03FDFC]/30",
    "transition-colors duration-200"
  );

  return (
    <div className="w-full max-w-md mx-auto">
      {/* Glass card */}
      <div className="bg-white/5 backdrop-blur-lg border border-white/10 rounded-3xl p-6 shadow-2xl">

        {/* Toggle tabs */}
        <div className="flex gap-1 bg-white/5 rounded-xl p-1 mb-6">
          {(["login", "signup"] as Mode[]).map((m) => (
            <button
              key={m}
              type="button"
              onClick={() => switchMode(m)}
              className={cn(
                "flex-1 py-2 rounded-lg text-sm font-medium transition-all duration-200",
                mode === m
                  ? "bg-white/10 text-white shadow-sm"
                  : "text-white/40 hover:text-white/60"
              )}
              style={{ fontFamily: "var(--font-satoshi)" }}
            >
              {m === "login" ? "Log In" : "Sign Up"}
            </button>
          ))}
        </div>

        {/* ── Sign Up Form ── */}
        {mode === "signup" && (
          <form onSubmit={handleSignup} className="flex flex-col gap-3">
            <div className="grid grid-cols-2 gap-3">
              <div className="flex flex-col gap-1.5">
                <label className="text-xs text-white/50 px-1" style={{ fontFamily: "var(--font-dm-sans)" }}>
                  First Name
                </label>
                <input
                  required
                  type="text"
                  placeholder="Ada"
                  value={firstName}
                  onChange={(e) => setFirstName(e.target.value)}
                  className={inputClass}
                  style={{ fontFamily: "var(--font-dm-sans)" }}
                />
              </div>
              <div className="flex flex-col gap-1.5">
                <label className="text-xs text-white/50 px-1" style={{ fontFamily: "var(--font-dm-sans)" }}>
                  Last Name
                </label>
                <input
                  required
                  type="text"
                  placeholder="Lovelace"
                  value={lastName}
                  onChange={(e) => setLastName(e.target.value)}
                  className={inputClass}
                  style={{ fontFamily: "var(--font-dm-sans)" }}
                />
              </div>
            </div>

            <div className="flex flex-col gap-1.5">
              <label className="text-xs text-white/50 px-1" style={{ fontFamily: "var(--font-dm-sans)" }}>
                Email
              </label>
              <input
                required
                type="email"
                placeholder="ada@example.com"
                value={signupEmail}
                onChange={(e) => setSignupEmail(e.target.value)}
                className={inputClass}
                style={{ fontFamily: "var(--font-dm-sans)" }}
              />
            </div>

            <div className="flex flex-col gap-1.5">
              <label className="text-xs text-white/50 px-1" style={{ fontFamily: "var(--font-dm-sans)" }}>
                Password
              </label>
              <div className="relative">
                <input
                  required
                  type={showPassword ? "text" : "password"}
                  placeholder="Min. 8 characters"
                  value={signupPassword}
                  onChange={(e) => setSignupPassword(e.target.value)}
                  className={cn(inputClass, "pr-11")}
                  style={{ fontFamily: "var(--font-dm-sans)" }}
                />
                <button
                  type="button"
                  onClick={() => setShowPassword(!showPassword)}
                  className="absolute right-3 top-1/2 -translate-y-1/2 text-white/30 hover:text-white/60 transition-colors"
                  tabIndex={-1}
                >
                  {showPassword ? <EyeOff className="w-4 h-4" /> : <Eye className="w-4 h-4" />}
                </button>
              </div>
            </div>

            <div className="flex flex-col gap-1.5">
              <label className="text-xs text-white/50 px-1" style={{ fontFamily: "var(--font-dm-sans)" }}>
                Confirm Password
              </label>
              <div className="relative">
                <input
                  required
                  type={showConfirm ? "text" : "password"}
                  placeholder="Repeat password"
                  value={confirmPassword}
                  onChange={(e) => setConfirmPassword(e.target.value)}
                  className={cn(inputClass, "pr-11")}
                  style={{ fontFamily: "var(--font-dm-sans)" }}
                />
                <button
                  type="button"
                  onClick={() => setShowConfirm(!showConfirm)}
                  className="absolute right-3 top-1/2 -translate-y-1/2 text-white/30 hover:text-white/60 transition-colors"
                  tabIndex={-1}
                >
                  {showConfirm ? <EyeOff className="w-4 h-4" /> : <Eye className="w-4 h-4" />}
                </button>
              </div>
            </div>

            <button
              type="submit"
              disabled={isLoading}
              className={cn(
                "mt-2 w-full h-12 rounded-xl font-semibold text-sm text-black",
                "bg-[#03FDFC] shadow-[0_0_15px_rgba(3,253,252,0.3)]",
                "hover:shadow-[0_0_24px_rgba(3,253,252,0.45)]",
                "disabled:opacity-60 disabled:cursor-not-allowed",
                "transition-shadow duration-200 flex items-center justify-center gap-2"
              )}
              style={{ fontFamily: "var(--font-satoshi)" }}
            >
              {isLoading ? (
                <>
                  <Loader2 className="w-4 h-4 animate-spin" />
                  Creating account…
                </>
              ) : (
                "Create Account"
              )}
            </button>
          </form>
        )}

        {/* ── Log In Form ── */}
        {mode === "login" && (
          <form onSubmit={handleLogin} className="flex flex-col gap-3">
            <div className="flex flex-col gap-1.5">
              <label className="text-xs text-white/50 px-1" style={{ fontFamily: "var(--font-dm-sans)" }}>
                Email
              </label>
              <input
                required
                type="text"
                placeholder="ada@example.com"
                value={identifier}
                onChange={(e) => setIdentifier(e.target.value)}
                className={inputClass}
                style={{ fontFamily: "var(--font-dm-sans)" }}
              />
            </div>

            <div className="flex flex-col gap-1.5">
              <label className="text-xs text-white/50 px-1" style={{ fontFamily: "var(--font-dm-sans)" }}>
                Password
              </label>
              <div className="relative">
                <input
                  required
                  type={showPassword ? "text" : "password"}
                  placeholder="Your password"
                  value={loginPassword}
                  onChange={(e) => setLoginPassword(e.target.value)}
                  className={cn(inputClass, "pr-11")}
                  style={{ fontFamily: "var(--font-dm-sans)" }}

                <button
```

## Task 8: Update WalletBuilder — localStorage user_id + VirtualCard integration

**Files:**
- Modify: `src/components/WalletBuilder.tsx`

- [ ] **Step 1: Replace the entire WalletBuilder.tsx**

Replace `src/components/WalletBuilder.tsx` with:

```tsx
"use client";

import { useState, useEffect, useRef } from "react";
import { useDebounce } from "use-debounce";
import {
  Search,
  X,
  Loader2,
  Wallet,
  ChevronRight,
} from "lucide-react";
import { toast } from "sonner";
import { Input } from "@/components/ui/input";
import { Button } from "@/components/ui/button";
import { VirtualCard } from "@/components/ui/VirtualCard";
import { cn } from "@/lib/utils";

interface Card {
  card_id: number;
  bank_name: string;
  card_name: string;
}

const API_BASE = "http://localhost:5000";

function getUserId(): string {
  if (typeof window === "undefined") return "user_123";
  return localStorage.getItem("zuno_user_id") ?? "user_123";
}

export default function WalletBuilder() {
  const [query, setQuery] = useState("");
  const [debouncedQuery] = useDebounce(query, 350);
  const [searchResults, setSearchResults] = useState<Card[]>([]);
  const [selectedCards, setSelectedCards] = useState<Card[]>([]);
  const [isSearching, setIsSearching] = useState(false);
  const [isSaving, setIsSaving] = useState(false);
  const [resultsKey, setResultsKey] = useState(0);
  const inputRef = useRef<HTMLInputElement>(null);

  useEffect(() => {
    if (!debouncedQuery.trim()) {
      setSearchResults([]);
      return;
    }

    const search = async () => {
      setIsSearching(true);
      try {
        const res = await fetch(
          `${API_BASE}/api/cards/search?q=${encodeURIComponent(debouncedQuery)}`
        );
        if (!res.ok) throw new Error();
        const data: Card[] = await res.json();
        setSearchResults(data);
        setResultsKey((k) => k + 1);
      } catch {
        toast.error("Could not reach the card database.");
        setSearchResults([]);
      } finally {
        setIsSearching(false);
      }
    };

    search();
  }, [debouncedQuery]);

  const toggleCard = (card: Card) => {
    setSelectedCards((prev) => {
      const exists = prev.find((c) => c.card_id === card.card_id);
      if (exists) return prev.filter((c) => c.card_id !== card.card_id);
      return [...prev, card];
    });
  };

  const isSelected = (card_id: number) =>
    selectedCards.some((c) => c.card_id === card_id);

  const removeCard = (card_id: number) => {
    setSelectedCards((prev) => prev.filter((c) => c.card_id !== card_id));
  };

  const saveWallet = async () => {
    if (selectedCards.length === 0) {
      toast.error("Select at least one card first.");
      return;
    }

    setIsSaving(true);
    try {
      const res = await fetch(`${API_BASE}/api/user/wallet`, {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          user_id: getUserId(),
          card_ids: selectedCards.map((c) => c.card_id),
        }),
      });

      if (!res.ok) throw new Error();
      toast.success(
        `${selectedCards.length} card${selectedCards.length !== 1 ? "s" : ""} saved to your wallet.`
      );
    } catch {
      toast.error("Save failed — is the backend running on port 5000?");
    } finally {
      setIsSaving(false);
    }
  };

  const clearSearch = () => {
    setQuery("");
    setSearchResults([]);
    inputRef.current?.focus();
  };

  return (
    <div className="w-full grid grid-cols-1 lg:grid-cols-[1fr_360px] gap-6 items-start">

      {/* ── Left panel: Search ── */}
      <div className="flex flex-col gap-4">

        {/* Search input */}
        <div className="relative group">
          <div className="absolute left-4 top-1/2 -translate-y-1/2 text-[--muted-foreground] group-focus-within:text-[--gold] transition-colors duration-200">
            {isSearching ? (
              <Loader2 className="w-4 h-4 animate-spin" />
            ) : (
              <Search className="w-4 h-4" />
            )}
          </div>

          <Input
            ref={inputRef}
            value={query}
            onChange={(e) => setQuery(e.target.value)}
            placeholder="Search by card name or bank…"
            className={cn(
              "pl-11 pr-11 h-13 bg-[--surface] border-[--border]",
              "text-[--foreground] placeholder:text-[--muted-foreground]",
              "focus-visible:ring-[--gold] focus-visible:ring-1 focus-visible:border-[--gold]/40",
              "rounded-xl text-sm transition-all duration-200",
              "hover:border-white/15"
            )}
          />

          {query && (
            <button
              onClick={clearSearch}
              className="absolute right-4 top-1/2 -translate-y-1/2 text-[--muted-foreground] hover:text-[--foreground] transition-colors"
              aria-label="Clear search"
            >
              <X className="w-4 h-4" />
            </button>
          )}
        </div>

        {/* Search results — VirtualCard grid */}
        {searchResults.length > 0 && (
          <div>
            <p className="text-xs text-[--muted-foreground] uppercase tracking-widest mb-3 px-1">
              {searchResults.length} result{searchResults.length !== 1 ? "s" : ""}
            </p>
            <div
              key={resultsKey}
              className="card-stagger grid grid-cols-1 sm:grid-cols-2 gap-3"
            >
              {searchResults.map((card) => (
                <VirtualCard
                  key={card.card_id}
                  cardName={card.card_name}
                  bankName={card.bank_name}
                  selected={isSelected(card.card_id)}
                  onClick={() => toggleCard(card)}
                />
              ))}
            </div>
          </div>
        )}

        {/* Empty state */}
        {!query && searchResults.length === 0 && (
          <div className="flex flex-col items-center justify-center py-16 text-center">
            <div className="w-12 h-12 rounded-2xl bg-[--surface] border border-[--border] flex items-center justify-center mb-4">
              <Search className="w-5 h-5 text-[--muted-foreground]" />
            </div>
            <p className="text-sm text-[--muted-foreground] max-w-xs leading-relaxed" style={{ fontFamily: "var(--font-dm-sans)" }}>
              Search for cards by name or bank to add them to your wallet.
            </p>
          </div>
        )}

        {/* No results */}
        {query && !isSearching && searchResults.length === 0 && (
          <div className="flex flex-col items-center justify-center py-12 text-center animate-slide-up">
            <p className="text-sm text-[--muted-foreground]" style={{ fontFamily: "var(--font-dm-sans)" }}>
              No cards found for &ldquo;{query}&rdquo;
            </p>
          </div>
        )}
      </div>

      {/* ── Right panel: Wallet ── */}
      <div className="lg:sticky lg:top-8 flex flex-col gap-4">
        <div className="rounded-2xl border border-[--border] bg-[--surface] overflow-hidden">

          {/* Wallet header */}
          <div className="flex items-center justify-between px-5 py-4 border-b border-[--border]">
            <div className="flex items-center gap-2.5">
              <Wallet className="w-4 h-4 text-[--gold]" />
              <span
                className="text-sm font-semibold tracking-wide text-[--foreground]"
                style={{ fontFamily: "var(--font-satoshi)" }}
              >
                My Wallet
              </span>
            </div>
            {selectedCards.length > 0 && (
              <span className="text-xs bg-[--gold]/15 text-[--gold] px-2.5 py-1 rounded-full font-medium">
                {selectedCards.length}
              </span>
            )}
          </div>

          {/* Card list — compact VirtualCards */}
          <div className="p-3">
            {selectedCards.length === 0 ? (
              <div className="flex flex-col items-center justify-center py-10 text-center">
                <div className="w-10 h-10 rounded-xl bg-[--surface-elevated] flex items-center justify-center mb-3">
                  <Wallet className="w-4 h-4 text-[--muted-foreground]" />
                </div>
                <p className="text-xs text-[--muted-foreground] leading-relaxed max-w-[180px]" style={{ fontFamily: "var(--font-dm-sans)" }}>
                  Cards you select will appear here.
                </p>
              </div>
            ) : (
              <div
                key={selectedCards.length}
                className="wallet-stagger flex flex-col gap-2"
              >
                {selectedCards.map((card) => (
                  <div key={card.card_id} className="relative group">
                    <VirtualCard
                      cardName={card.card_name}
                      bankName={card.bank_name}
                      compact
                    />
                    {/* Remove button overlay */}
                    <button
                      onClick={() => removeCard(card.card_id)}
                      className={cn(
                        "absolute top-2 right-2 w-6 h-6 rounded-full",
                        "bg-black/40 backdrop-blur-sm",
                        "flex items-center justify-center",
                        "opacity-0 group-hover:opacity-100",
                        "text-white/50 hover:text-red-400 hover:bg-red-500/20",
                        "transition-all duration-150"
                      )}
                      aria-label={`Remove ${card.card_name}`}
                    >
                      <X className="w-3 h-3" />
                    </button>
                  </div>
                ))}
              </div>
            )}
          </div>
        </div>

        {/* Save button */}
        <Button
          onClick={saveWallet}
          disabled={isSaving || selectedCards.length === 0}
          className={cn(
            "w-full h-12 rounded-xl font-semibold text-sm tracking-wide",
            "transition-all duration-200",
            selectedCards.length > 0
              ? "bg-[--gold] hover:bg-[--gold]/90 text-[#07070D] shadow-[0_0_24px_rgba(201,168,76,0.2)] hover:shadow-[0_0_32px_rgba(201,168,76,0.3)]"
              : "bg-[--surface] text-[--muted-foreground] border border-[--border] cursor-not-allowed"
          )}
          style={{ fontFamily: "var(--font-satoshi)" }}
        >
          {isSaving ? (
            <span className="flex items-center gap-2">
              <Loader2 className="w-4 h-4 animate-spin" />
              Saving…
            </span>
          ) : (
            <span className="flex items-center gap-2">
              Save Wallet
              <ChevronRight className="w-4 h-4" />
            </span>
          )}
        </Button>

        {selectedCards.length > 0 && (
          <p className="text-center text-xs text-[--muted-foreground]" style={{ fontFamily: "var(--font-dm-sans)" }}>
            {selectedCards.length} card{selectedCards.length !== 1 ? "s" : ""} selected
          </p>
        )}
      </div>
    </div>
  );
}
```
